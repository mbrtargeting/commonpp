project(test C CXX)
cmake_minimum_required(VERSION 3.0)

add_definitions("-std=c++11")

set(CMAKE_MODULE_PATH
    "${CMAKE_MODULE_PATH};${test_SOURCE_DIR}")

#Thread
find_package(Threads REQUIRED)
find_package(TBB REQUIRED)
include_directories(${TBB_include_DIRS})

#Boost
if (${BUILD_SHARED_LIBS})
    set(Boost_USE_STATIC_LIBS    OFF)
    add_definitions("-DBOOST_ALL_DYN_LINK=1")
ELSE()
    set(Boost_USE_STATIC_LIBS    ON)
endif()

set(Boost_USE_MULTITHREADED  ON)
find_package(Boost 1.54.0 COMPONENTS iostreams unit_test_framework log log_setup filesystem thread timer chrono date_time program_options system REQUIRED)
include_directories(${Boost_include_DIR})
link_directories(${Boost_LIBRARY_DIR})

add_subdirectory(commonpp ${CMAKE_CURRENT_BINARY_DIR}/commonpp)
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/commonpp/include)
include_directories(${CMAKE_CURRENT_BINARY_DIR}/commonpp/include)

add_executable(main main.cpp)
target_link_libraries(main commonpp
    ${Boost_LIBRARIES}
    TBB::tbb
    ${CMAKE_THREAD_LIBS_INIT})
